```python
def load_exclusion_rules(root: Path) -> ExclusionRules: ...

def is_excluded(path: Path, *, root: Path, rules: ExclusionRules) -> bool:
    """Return True if `path` should be skipped by the packer."""

    """Walk `root` yielding .py files that pass all three filters."""
```
